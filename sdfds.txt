// Widget mz-logo
class mz_logo extends \Elementor\Widget_Base
{

    public function get_name()
    {
        return 'mz_logo';
    }

    public function get_title()
    {
        return esc_html__('لوگو سایت', 'textdomain');
    }

    public function get_icon()
    {
        return 'eicon-site-logo';
    }

    public function get_categories()
    {
        return ['mz-category'];
    }

    public function get_keywords()
    {
        return ['keyword', 'keyword'];
    }

    protected function register_controls()
    {

        $this->start_controls_section(
            'section_content',
            [
                'label' => esc_html__('محتوا', 'textdomain'),
                'tab' => \Elementor\Controls_Manager::TAB_CONTENT,
            ]
        );


        $this->add_control(
            'custom_logo',
            [
                'label' => esc_html__('لوگو دلخواه', 'textdomain'),
                'type' => \Elementor\Controls_Manager::SWITCHER,
                'label_on' => esc_html__('بله', 'textdomain'),
                'label_off' => esc_html__('خیر', 'textdomain'),
                'return_value' => 'yes',
                'default' => 'no',
            ]
        );

        $this->add_control(
            'logo_image',
            [
                'label' => esc_html__('انتخاب تصویر محصول', 'textdomain'),
                'type' => \Elementor\Controls_Manager::MEDIA,
                'condition' => [
                    'custom_logo' => 'yes',
                ],
                'default' => [
                    'url' => \Elementor\Utils::get_placeholder_image_src(),
                ],
            ]
        );

        $this->end_controls_section();

        $this->start_controls_section(
            'section_style',
            [
                'label' => esc_html__('استایل', 'textdomain'),
                'tab' => \Elementor\Controls_Manager::TAB_STYLE,
            ]
        );

        $this->add_control(
            'logo_width',
            [
                'label' => esc_html__('عرض لوگو', 'textdomain'),
                'type' => \Elementor\Controls_Manager::SLIDER,
                'size_units' => ['px', '%', 'custom'],
                'range' => [
                    'px' => [
                        'min' => 0,
                        'max' => 1000,
                        'step' => 1,
                    ],
                ],
                'default' => [
                    'unit' => '%',
                    'size' => 100,
                ],
                'selectors' => [
                    '{{WRAPPER}} .mz-logo' => 'width: {{SIZE}}{{UNIT}};',
                ],
            ]
        );

        $this->add_control(
            'logo_height',
            [
                'label' => esc_html__('ارتفاع لوگو', 'textdomain'),
                'type' => \Elementor\Controls_Manager::SLIDER,
                'size_units' => ['px', '%', 'custom'],
                'range' => [
                    'px' => [
                        'min' => 0,
                        'max' => 1000,
                        'step' => 1,
                    ],
                ],
                'default' => [
                    'unit' => '%',
                    'size' => 100,
                ],
                'selectors' => [
                    '{{WRAPPER}} .mz-logo' => 'height: {{SIZE}}{{UNIT}};',
                ],
            ]
        );

        $this->end_controls_section();
    }

    protected function render()
    {
        $settings = $this->get_settings_for_display();

        $logo_image = $settings['logo_image']['url'];
        if ($logo_image) {
            $logo_image = $logo_image;
        } else {
            $logo_image = get_template_directory_uri() . '/assets/images/mz-logo.png';
        }
?>
        <a href="<?php echo home_url(); ?>" class="mz-logo">
            <img src="<?php echo $logo_image; ?>">
        </a>
    <?php
    }
}
